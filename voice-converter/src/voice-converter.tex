\def\radius{2cm}
\def\colorwheelwidth{0.2cm}
\def\innerradius{\radius-0.5*\colorwheelwidth}
\def\minorticklen{0.2cm}
\def\majorticklen{0.3cm}
\def\maxticklen{0.4cm}
\def\labelrad{1.3cm}
\def\sqrthalf{0.707}

\newcommand{\drawswitch}[1]{
  \draw[fill=black] (0,0) circle (.5mm);
  \node
  at (0,2.5cm) {\fontsize{50}{0}\textbf{\sf\selectfont #1}};
  \node
  at (0,+0.8cm) {\fontsize{40}{0}\textbf{\sf\selectfont Off}};
  \node
  at (0,-0.8cm) {\fontsize{40}{0}\textbf{\sf\selectfont On}};
}

\begin{tikzpicture}[scale=4,line width=1pt]
  \begin{scope}[xshift=-6cm]
    \drawswitch{Power}
  \end{scope}

  % Power LED
  \draw[fill=black] (-6cm,1.8cm) circle (.5mm);

  \begin{scope}[xshift=25cm]
    \drawswitch{}
  \end{scope}

  \begin{scope}[xshift=7.5cm,yshift=3cm, line width=2.0mm, line cap=round]
    \draw (-5cm,0) -- (+5cm,0) node [midway, fill=white, align=center]
      {\sf\bf\fontsize{50}{0}\selectfont\textbf{\sf{ NF Modulation }}};
    \draw (-5cm,0.0cm) -- (-5cm,-0.4cm);
    \draw (+5cm,0.0cm) -- (+5cm,-0.4cm);
  \end{scope}

  \begin{scope}[xshift=18.5cm,yshift=3cm, line width=2.0mm, line cap=round]
    \draw (-5cm,0) -- (+7.5cm,0) node [midway, fill=white, align=center]
      {\sf\bf\fontsize{50}{0}\selectfont\textbf{\sf{ HF Modulation }}};
    \draw (-5cm,0.0cm) -- (-5cm,-0.4cm);
    \draw (+7.5cm,0.0cm) -- (+7.5cm,-0.4cm);
  \end{scope}

  \foreach \x / \label / \xtrashift in {0/Volume/-2cm,1/Depth/0,2/Hacker Pitch/0,3/Depth/1cm,4/Hacker Pitch/1cm} {

    \begin{scope}[xshift=\x*5cm+\xtrashift]
      % add subtle gray tone as background
      \shade[shading=radial, inner color=white, outer color=gray!15] (0,0) circle (\radius);

      % partial dial gauge outline
      \draw[black] (\sqrthalf*\radius,-\sqrthalf*\radius)
      arc[start angle=-45, end angle=225, radius=\radius];
      \draw[fill=black] (0,0) circle (.5mm);
      \node[draw, circle, inner sep=.2mm,
        label={[label distance=9cm,
            font=\sf\bf\fontsize{50}{0}\selectfont]90:\textbf{\sf\label}}]
      (anchor) at (0,0) {};

      % color wheel
      \iftrue
      \draw[shading=color wheel]
      (225:{\innerradius-.5*\colorwheelwidth})
      arc [start angle=225, delta angle=-270,
        radius={\innerradius-.5*\colorwheelwidth}] --++
      ({225-270}:\colorwheelwidth)
      arc [start angle={225-270}, delta angle=270,
        radius={\innerradius+.5*\colorwheelwidth}] --
      cycle;
      \fi

      % short thin line at every 3 degrees
      \foreach \x in {-45,-42,...,225}
      \draw[line width=0.5mm] (\x:\radius-\minorticklen) -- (\x:\radius);

      % long thin line at every 15 degrees
      \foreach \x in {-45,-30,...,225}
      \draw[line width=0.5mm] (\x:\radius-\majorticklen) -- (\x:\radius);

      % very long fat line at every 30 degrees
      \foreach \x in {-45,-15,...,225}
      \draw[line width=1.0mm] (\x:\radius-\maxticklen) -- (\x:\radius);

      % labels
      \node[scale=1.4, rotate=0, font=\fontsize{30}{0}\selectfont]
      at (225:\labelrad) {\textbf{\sf Min}};
      \node[scale=1.4, rotate=0, font=\fontsize{30}{0}\selectfont]
      at (-45:\labelrad) {\textbf{\sf Max}};
    \end{scope}
  }
\end{tikzpicture}
